function dadosPainel() {
 let login = assoc.attributes.user;
 let tipo = "";
 if(assoc.isOper() == false && assoc.isAssociate() == false) { 
 let operJson = JSON.parse(localStorage.getItem('operData')); 
 login = operJson.CODOPERADOR;
 tipo = "operador";
 }
 let url = acsphelper.envurl("controllerapi.acsp.com.br/acesso/id/"+login); 
 const requestInfo = {
 method: 'GET',
 headers: new Headers({
 'Content-type': 'application/json',
 'Authorization': acsphelper.hash(),
 })
 }; 
 fetch(url, requestInfo)
 .then(response => {
 if (response.ok) {
 return response.text();
 } else {
 throw new Error("não foi possível comentar");
 }
 })
 .then(data => { 
 let userJson = JSON.parse(localStorage.getItem('assocData')); 
 data = JSON.parse(data); 
 if(assoc.isOper() == undefined && assoc.isAssociate() == undefined) { 
 userJson.tipo = "colaborador";
 userJson.dados = {A1_CONTATO : data[0].NOME,A1_EMAIL : data[0].EMAIL, A1_CGC: data[0].CPF, A1_NOME:data[0].NOME}; 
 } 
 if(tipo == "operador") userJson.tipo = "operador";
 if(assoc.isAssociate()) userJson.tipo = "associado";
 userJson.dados.painel = {painelId: data[0]['ID'],dtalt: data[0]['DT_ALT']};
